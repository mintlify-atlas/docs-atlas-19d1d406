## Overview

Learn how to create sophisticated automation workflows by combining multiple Darwin tasks, integrating with external tools, and using advanced configuration options.

## Multi-Task Workflows

### Sequential Task Execution

Chain multiple browser automation tasks together:

```typescript
import { BrowserAgent } from '@darwin/sdk';

interface WorkflowStep {
  name: string;
  website: string;
  task: string;
  maxSteps: number;
}

async function runWorkflow(steps: WorkflowStep[]) {
  const results = [];
  
  for (const [index, step] of steps.entries()) {
    console.log(`\n=== Step ${index + 1}: ${step.name} ===`);
    
    const agent = new BrowserAgent({
      website: step.website,
      task: step.task,
      maxSteps: step.maxSteps,
      model: 'google/gemini-3-flash-preview',
      env: 'LOCAL',
      verbose: 1,
    });
    
    await agent.init();
    
    try {
      const result = await agent.execute();
      results.push({
        step: step.name,
        success: result.result.success,
        thoughts: result.thoughts,
        steps: result.steps,
      });
      
      console.log(`✓ ${step.name} completed in ${result.steps} steps`);
    } catch (error) {
      console.error(`✗ ${step.name} failed:`, error);
      results.push({
        step: step.name,
        success: false,
        error: error.message,
      });
      break; // Stop workflow on error
    } finally {
      await agent.close();
    }
  }
  
  return results;
}

// Define a multi-step workflow
const ecommerceWorkflow: WorkflowStep[] = [
  {
    name: 'Product Search',
    website: 'http://localhost:3000',
    task: 'Search for "laptop" and find a product under $1000',
    maxSteps: 20,
  },
  {
    name: 'Add to Cart',
    website: 'http://localhost:3000/products/123',
    task: 'Add the product to cart and verify it appears in the cart',
    maxSteps: 15,
  },
  {
    name: 'Checkout',
    website: 'http://localhost:3000/cart',
    task: 'Complete the checkout process with test credit card',
    maxSteps: 30,
  },
];

// Run the workflow
const results = await runWorkflow(ecommerceWorkflow);

// Analyze results
const allSuccessful = results.every(r => r.success);
console.log(`\nWorkflow ${allSuccessful ? 'completed' : 'failed'}`);
```

### Parallel Task Execution

Run multiple independent tasks simultaneously:

```typescript
async function runParallelTasks(tasks: WorkflowStep[]) {
  const promises = tasks.map(async (task) => {
    const agent = new BrowserAgent({
      website: task.website,
      task: task.task,
      maxSteps: task.maxSteps,
      model: 'google/gemini-3-flash-preview',
      env: 'BROWSERBASE', // Use BROWSERBASE for parallel execution
    });
    
    await agent.init();
    
    try {
      const result = await agent.execute();
      return {
        name: task.name,
        success: true,
        result,
      };
    } finally {
      await agent.close();
    }
  });
  
  return await Promise.all(promises);
}

// Run multiple tests in parallel
const parallelTests = [
  {
    name: 'Homepage Load Test',
    website: 'http://localhost:3000',
    task: 'Navigate the homepage and verify all sections load',
    maxSteps: 10,
  },
  {
    name: 'Search Functionality Test',
    website: 'http://localhost:3000',
    task: 'Test the search feature with various queries',
    maxSteps: 20,
  },
  {
    name: 'Login Flow Test',
    website: 'http://localhost:3000',
    task: 'Test the login flow with valid credentials',
    maxSteps: 15,
  },
];

const results = await runParallelTasks(parallelTests);
```

### Domain-Specific Agents

```typescript
interface AgentPersona {
  name: string;
  systemPrompt: string;
  focusAreas: string[];
}

const personas: Record<string, AgentPersona> = {
  shopper: {
    name: 'Online Shopper',
    systemPrompt: 'You are a typical online shopper looking for good deals...',
    focusAreas: ['pricing', 'product-info', 'checkout'],
  },
  developer: {
    name: 'Technical User',
    systemPrompt: 'You are a developer evaluating the site technically...',
    focusAreas: ['performance', 'functionality', 'errors'],
  },
  senior: {
    name: 'Senior User',
    systemPrompt: 'You are a senior citizen less familiar with technology...',
    focusAreas: ['simplicity', 'clarity', 'accessibility'],
  },
};

async function testWithPersona(personaName: string, task: string) {
  const persona = personas[personaName];
  
  const agent = new BrowserAgent({
    website: 'http://localhost:3000',
    task,
    systemPrompt: persona.systemPrompt,
    maxSteps: 40,
  });
  
  await agent.init();
  try {
    const result = await agent.execute();
    return {
      persona: persona.name,
      result,
    };
  } finally {
    await agent.close();
  }
}
```

## Integration with External Tools

### Webhook Notifications

Send results to external services:

```typescript
import fetch from 'node-fetch';

async function runWithWebhook(config: any, webhookUrl: string) {
  const agent = new BrowserAgent(config);
  await agent.init();
  
  try {
    const result = await agent.execute();
    
    // Send results to webhook
    await fetch(webhookUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        timestamp: new Date().toISOString(),
        task: config.task,
        success: result.result.success,
        steps: result.steps,
        thoughts: result.thoughts,
      }),
    });
    
    return result;
  } finally {
    await agent.close();
  }
}
```

### Slack Integration

```typescript
import { WebClient } from '@slack/web-api';

const slack = new WebClient(process.env.SLACK_TOKEN);

async function runWithSlackNotification(config: any, channel: string) {
  const agent = new BrowserAgent(config);
  await agent.init();
  
  try {
    const startTime = Date.now();
    const result = await agent.execute();

      blocks: [
        {

          text: {

          },

        {
          type: 'section',
          text: {

    });

  } finally {
    await agent.close();
  }
}
```

### Database Logging

```typescript
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

async function runWithDatabaseLogging(config: any) {
  const agent = new BrowserAgent(config);
  await agent.init();
  
  // Create run record
  const run = await prisma.automationRun.create({
    data: {
      website: config.website,
      task: config.task,
      startedAt: new Date(),
      status: 'running',
    },
  });
  
  try {
    const result = await agent.execute();
    
    // Update run record
    await prisma.automationRun.update({
      where: { id: run.id },
      data: {
        status: result.result.success ? 'success' : 'failed',
        completedAt: new Date(),
        steps: result.steps,
        thoughts: {
          create: result.thoughts.map(thought => ({
            step: thought.step,
            text: thought.text,
            timestamp: new Date(thought.timestamp),
            source: thought.source,
          })),
        },
      },
    });
    
    return result;
  } catch (error) {
    await prisma.automationRun.update({
      where: { id: run.id },
      data: {
        status: 'error',
        error: error.message,
        completedAt: new Date(),
      },
    });
    throw error;
  } finally {
    await agent.close();
  }
}
```

## Event-Driven Workflows

### Real-Time Event Handling

```typescript
import { BrowserAgent } from '@darwin/sdk';
import EventEmitter from 'events';

class WorkflowOrchestrator extends EventEmitter {
  async runTask(config: any) {
    const agent = new BrowserAgent({
      ...config,
      onEvent: (type, data) => {
        // Emit events for external listeners
        this.emit(`agent:${type}`, data);
      },
    });
    
    await agent.init();
    
    try {
      const result = await agent.execute();
      this.emit('task:complete', { config, result });
      return result;
    } catch (error) {
      this.emit('task:error', { config, error });
      throw error;
    } finally {
      await agent.close();
    }
  }
}

const orchestrator = new WorkflowOrchestrator();

// Listen to events
orchestrator.on('agent:think', (data) => {
  console.log('💭 Agent thinking:', data);
});

orchestrator.on('agent:action', (data) => {
  console.log('🔧 Agent action:', data.toolName);
});

orchestrator.on('task:complete', ({ config, result }) => {
  console.log(`✓ Task "${config.task}" completed`);
});

orchestrator.on('task:error', ({ config, error }) => {
  console.error(`✗ Task "${config.task}" failed:`, error.message);
});

// Run tasks
await orchestrator.runTask({
  website: 'http://localhost:3000',
  task: 'Test the checkout flow',
  maxSteps: 30,
});
```

## Advanced Configuration

### Dynamic Configuration

```typescript
interface DynamicConfig {
  environment: 'development' | 'staging' | 'production';
  userType: 'guest' | 'authenticated';
  viewport: 'mobile' | 'tablet' | 'desktop';
}

function buildAgentConfig(params: DynamicConfig) {
  const baseUrl = {
    development: 'http://localhost:3000',
    staging: 'https://staging.example.com',
    production: 'https://example.com',
  }[params.environment];
  
  const viewportSizes = {
    mobile: { width: 375, height: 667 },
    tablet: { width: 768, height: 1024 },
    desktop: { width: 1920, height: 1080 },
  };
  
  return {
    website: baseUrl,
    task: params.userType === 'guest' 
      ? 'Browse products as a guest user'
      : 'Browse products as a logged-in user',
    viewport: viewportSizes[params.viewport],
    maxSteps: 30,
  };
}

// Use dynamic config
const config = buildAgentConfig({
  environment: 'staging',
  userType: 'guest',
  viewport: 'mobile',
});

const agent = new BrowserAgent(config);
```

## Best Practices

<Tip>
**Modular Tasks**: Break complex workflows into smaller, reusable tasks.
</Tip>

<Tip>
**Error Handling**: Always implement proper error handling and cleanup in workflows.
</Tip>

<Tip>
**Event Logging**: Use the `onEvent` callback to log and monitor agent behavior in real-time.
</Tip>

<Tip>
**Configuration Management**: Use environment variables and config files for flexible deployment.
</Tip>

<Tip>
**Resource Cleanup**: Always call `agent.close()` in a `finally` block to prevent resource leaks.
</Tip>

<Warning>
When running parallel tasks with BROWSERBASE, be mindful of rate limits and costs.
</Warning>
